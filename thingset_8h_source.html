141 /*
142  * Internal access rights to data objects
143  */
144 #define TS_READ_ALL (0x1U << 0) // read access for all
145 #define TS_READ_USER (0x3U << 0) // read after authentication as normal user
146 #define TS_READ_MAKER (0x7U << 0) // read after authentication as manufacturer, e.g. for factory calibration
147 #define TS_READ_MASK TS_READ_MAKER // maker has all 3 bits set
148 
149 #define TS_WRITE_ALL (0x1U << 3) // write access for all
150 #define TS_WRITE_USER (0x3U << 3) // write after authentication as normal user
151 #define TS_WRITE_MAKER (0x7U << 3) // write after authentication as manufacturer, e.g. for factory calibration
152 #define TS_WRITE_MASK TS_WRITE_MAKER
153 
154 #define TS_EXEC_ALL (0x1U << 6) // execute access for all (only for RPC)
155 #define TS_EXEC_USER (0x3U << 6) // execute after authentication as normal user
156 #define TS_EXEC_MAKER (0x7U << 6) // execute after authentication as manufacturer, e.g. for factory calibration
157 #define TS_EXEC_MASK TS_EXEC_MAKER
